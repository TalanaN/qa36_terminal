14. Сделать текстовый файл tf_4.txt, в котором будет 15 строк.
(a)
	cat > tf_4.sh <<EOF
	#!/bin/bash
	for (( i=1 ; i<=15 ; i++ )); do
	echo "this is $i string";
	done >> tf_4.txt
	EOF
	./tf_4.sh
(b)
	cat > tf_4.sh <<EOF
	#!/bin/bash
	for i in {1..15}; do
	echo "this is $i string";
	done >> tf_4.txt
	EOF
	./tf_4.sh
(c)
	cat > tf_4.sh <<EOF
	#!/bin/bash
	i=1
	while [[ i -le 15 ]] ; do
	echo "this is $i string";
	(( i += 1 ))
	done >> tf_4.txt
	EOF
	./tf_4.sh
